def create
  if current_user.checkins.create! params[:checkin]
    respond_to do |format|
      format.json { render :json => {"status" => "Successful checkin." }}
    end
  else
    respond_to do |format|
      format.json {render :json => {"status" => "Fail.", "error" => "couldn't check in"} }
    end
  end
end
